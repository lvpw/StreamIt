         BUFFER[push_index++] = ... ;
       }                                
     }                                  
                                        
     work {                             
       // run Source                    
       BUFFER[push_index] = ... ;
       push_index = (push_index+1) & 3;              
                                        
       // run FIR                       
       float result = 0;                
       for (int i=1; i<PEEK; i++) {     
         result += i*BUFFER[(pop_index+i) & 3];    
       }                                
       pop_index = (pop_index+1) & 3;     
       print(result);                   
     }                                  
   }
